data.columns = ['label', 'text', 'unused1', 'unused2', 'unused3']

data = data[['label', 'text']]

data['text'] = data['text'].str.lower()

def remove_accents(text):
    text = re.sub(r'[]', 'a', text)
    text = re.sub(r'[]', 'e', text)
    text = re.sub(r'[]', 'i', text)
    text = re.sub(r'[]', 'o', text)
    text = re.sub(r'[]', 'u', text)
    return text

data['text'] = data['text'].apply(remove_accents)

data['text'] = data['text'].apply(lambda x: re.sub(r'[^a-zA-Z\s]', '', x))

data['text'] = data['text'].apply(word_tokenize)

stop_words = set(stopwords.words('english'))
data['text'] = data['text'].apply(lambda x: [word for word in x if word not in stop_words])

lemmatizer = WordNetLemmatizer()
data['text'] = data['text'].apply(lambda x: [lemmatizer.lemmatize(word) for word in x])
data['text'] = data['text'].apply(lambda x: [lemmatizer.lemmatize(word) for word in x])

data['text'] = data['text'].apply(lambda x: ' '.join(x))

def show_metrics(model, X_test, y_test):
    y_pred = model.predict(X_test)
    cm = confusion_matrix(y_test, y_pred)
    precision = precision_score(y_test, y_pred, pos_label='spam')
    recall = recall_score(y_test, y_pred, pos_label='spam')

    print("Matriz de Confusin:")
    print(cm)
    print("Precision:", precision)
    print("Recall:", recall)

    tp_ham = cm[0][0]
    fn_ham = cm[0][1]
    fp_ham = cm[1][0]

    precision_ham = tp_ham / (tp_ham + fp_ham)
    recall_ham = tp_ham / (tp_ham + fn_ham)

    tp_spam = cm[1][1]
    fn_spam = cm[1][0]
    fp_spam = cm[0][1]

    precision_spam = tp_spam / (tp_spam + fp_spam)
    recall_spam = tp_spam / (tp_spam + fn_spam)

    print("Precisin (ham):", precision_ham)
    print("Recall (ham):", recall_ham)
    print("Precisin (spam):", precision_spam)
    print("Recall (spam):", recall_spam)

    y_scores = model.predict_proba(X_test)[:, 1]
    fpr, tpr, thresholds = roc_curve(y_test, y_scores, pos_label='spam')
    roc_auc = auc(fpr, tpr)

    plt.figure()
    plt.plot(fpr, tpr, color='darkorange', lw=2, label='ROC curve (area = %0.2f)' % roc_auc)
    plt.plot([0, 1], [0, 1], color='navy', lw=2, linestyle='--')
    plt.xlim([0.0, 1.0])
    plt.ylim([0.0, 1.05])
    plt.xlabel('False Positive Rate')
    plt.ylabel('True Positive Rate')
    plt.title('Receiver Operating Characteristic')
    plt.legend(loc="lower right")
    plt.show()

1. Qu error es ms aceptable: dejar pasar un SMS de SPAM (falso negativo) o bloquear un SMS legtimo (falso positivo)? Justifique su respuesta.

Generalmente el impacto de dejar pasar un SMS de spam (falso negativo) tiende a ser ms perjudicial que el de bloquear un SMS legtimo (falso positivo). Esto se debe a que el spam puede contener contenido malicioso que puede afectar la seguridad del usuario, mientras que un SMS legtimo bloqueado puede ser recuperado por el usuario.

Adems de eso un SMS de spam que no es detectado y pasa desapercibido puede llevar a que el usuario sea engaado o vctima de estafas, phishing u otras actividades fraudulentas. Por otro lado bloquear un SMS legtimo puede ser molesto para el usuario, pero dejar pasar un SMS de spam puede ser percibido como una violacin ms grave de la privacidad, ya que implica recibir mensajes no deseados que podran contener contenido ofensivo, engaoso o malicioso.

Aplicacion de umbrales

En los procesos de vectorizacin de texto, tanto en TfidfVectorizer como en CountVectorizer, los umbrales min_df y max_df sirven para filtrar tokens basndose en su frecuencia de aparicin en el conjunto de documentos. El parmetro min_df establece el nmero mnimo de documentos en los que un token debe aparecer para ser incluido en la matriz de caractersticas, ayudando a eliminar trminos raros que pueden no aportar valor significativo. Por otro lado, max_df define el lmite mximo de documentos en los que un token puede aparecer; los tokens que superen este umbral se consideran demasiado comunes y se excluyen, con el fin de eliminar palabras que son tan frecuentes que no ayudan a diferenciar entre textos. Estos umbrales ayudan a optimizar la seleccin de caractersticas, mejorando as la relevancia de los datos para el modelado y la eficiencia computacional
